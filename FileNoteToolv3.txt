function QFCCancellationCopy() {
	var d= new Date();
	var hours = d.getHours();
	var minutes = d.getMinutes();
	var ampm = hours >= 12 ? 'pm' : 'am';
	hours = hours % 12;
	hours = hours ? hours : 12;
	minutes = minutes < 10 ? '0'+minutes : minutes;
	var TwelveHourTime = hours + ':' + minutes + ' ' + ampm;

	document.getElementById("CurrentDateAndTime").innerHTML=(d.getMonth()+1) +"/"+ d.getDate() +"/"+ d.getFullYear() +" "+TwelveHourTime
	document.getElementById("JurisdictionStateMimic").innerHTML=document.getElementById("JurisdictionState").value
	document.getElementById("claimownerMimic").innerHTML=document.getElementById("claimowner").value
	document.getElementById("CancellationReasonMimic").innerHTML=document.getElementById("CancellationReason").value
	document.getElementById("QFCCancellationCommentsMimic").innerHTML=document.getElementById("QFCCancellationComments").value

	var div = document.getElementById('NoContactFilenote');
	div.contentEditable = 'true';
	var controlRange;
	if (document.body.createControlRange) {
	controlRange = document.body.createControlRange();
	controlRange.addElement(div);
	controlRange.execCommand('Copy');
	}
	div.contentEditable = 'false';
}

function copyTextToClipboard() {
  var claimNum = document.getElementById('claimnum').innerHTML;
  var nlFilenote = 	'Inquiry from: '+document.getElementById('caller').value+'\n'+
							'Regarding: '+document.getElementById('inquiry').value+'\n'+
							'Resolution: '+document.getElementById('resolution').value+'\n'
  var lbFilenote = nlFilenote.replace(/(?:\r\n|\r|\n)/g, '<br />');					

  //INFO Below from http://stackoverflow.com/questions/400212/how-do-i-copy-to-the-clipboard-in-javascript
  var textArea = document.createElement("textarea");

  //
  // *** This styling is an extra step which is likely not required. ***
  //
  // Why is it here? To ensure:
  // 1. the element is able to have focus and selection.
  // 2. if element was to flash render it has minimal visual impact.
  // 3. less flakyness with selection and copying which **might** occur if
  //    the textarea element is not visible.
  //
  // The likelihood is the element won't even render, not even a flash,
  // so some of these are just precautions. However in IE the element
  // is visible whilst the popup box asking the user for permission for
  // the web page to copy to the clipboard.
  //

  // Place in top-left corner of screen regardless of scroll position.
  textArea.style.position = 'fixed';
  textArea.style.top = 0;
  textArea.style.left = 0;

  // Ensure it has a small width and height. Setting to 1px / 1em
  // doesn't work as this gives a negative w/h on some browsers.
  textArea.style.width = '2em';
  textArea.style.height = '2em';

  // We don't need padding, reducing the size if it does flash render.
  textArea.style.padding = 0;

  // Clean up any borders.
  textArea.style.border = 'none';
  textArea.style.outline = 'none';
  textArea.style.boxShadow = 'none';

  // Avoid flash of white box if rendered for any reason.
  textArea.style.background = 'transparent';


  textArea.value = nlFilenote;

  document.body.appendChild(textArea);

  textArea.select();

  try {
    var successful = document.execCommand('copy');
    var msg = successful ? 'successful' : 'unsuccessful';
    <!-- alert("Copying text command was " + msg); -->
  } catch (err) {
    alert("Fatal Error")
  }

  document.body.removeChild(textArea);
  // document.getElementById("FileNoteDiv").style.visibility="hidden";
};
